from functools import lru_cache

@lru_cache(maxsize=128)
def fibonacci(n):
    """Return the nth Fibonacci number using memoization."""
    if n < 2:
        return n
    return fibonacci(n-1) + fibonacci(n-2)

def print_fib_sequence(limit):
    sequence = [fibonacci(i) for i in range(limit)]
    print(f"Fibonacci sequence up to {limit}:", sequence)

print_fib_sequence(15)

import re
from collections import Counter

def analyze_text(text):
    """Analyze text: count words, sentences, and most common words."""
    words = re.findall(r'\w+', text.lower())
    sentences = re.split(r'[.!?]+', text)
    word_count = len(words)
    sentence_count = len([s for s in sentences if s.strip()])
    most_common = Counter(words).most_common(3)
    print(f"Word count: {word_count}")
    print(f"Sentence count: {sentence_count}")
    print("Most common words:")
    for word, count in most_common:
        print(f"  {word}: {count}")

sample = "Hello world! This is a sample sentence. Hello again! How many words are here? This is just a test."
analyze_text(sample)
